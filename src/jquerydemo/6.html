选择器 实例 选取
* $("*") 所有元素
#id $("#lastname") id="lastname" 的元素
element $("td") 所有td的元素
.class $(".firstname") class="firstname"
selector1,selector2,...selectorN $("p,div") 所有p标签和div标签的元素
ancestor descendant $("tr td") tr下的所有td元素
parent > child $("tr>td") tr下子元素td
prev + next $("div+p") 匹配所有div后面的p元素
prev ~ siblings $("div~div") 匹配所有div的同辈元素div元素
:first $("div p:first") 获取div中第一个p元素
:last $("p:last") 匹配所有p元素中最后一个元素
:not $("input:not(:checked)") 查找所有未选中的input元素
:odd $("tr:odd") 查找tr中索引值为奇数的
:even $("tr:even") 查找tr中索引值为偶数的
:eq(index) $("p:eq(0)") 匹配索引值为0的p元素
:gt(index) $("p:gt(0)") 匹配大于0的索引值为p的元素
:lt(index) $("p:lt()") 匹配小于0的索引值为p的元素
:header $(":header") 匹配h1 h2 h3之类的元素
:animated $(":animated") 匹配所有正在执行动画效果的元素
:focus $(":focus") 匹配当前所有获取焦点的元素
:contains(text) $("div:contains("John")") 查找所有包含 "John" 的 div 元素
:empty $("td:empty") 查找所有不包含子元素或者文本的空元素
:has(selector) $("div:has(p)") 匹配所有包含p元素的div元素
:hidden $("tr:hidden") 匹配所有隐藏的tr元素
:visible $("tr:visible") 匹配所有可见的tr元素
[attribute] $("div[id]") 查找所有含有 id 属性的 div 元素
[attribute=value] $("input[name='newsletter']") 查找所有 name 属性是 newsletter 的 input 元素
[attribute^=value] $("input[name^='news']") 查找所有 name 以 'news' 开始的 input 元素
[attribute$=value] $("input[name$='letter']") 查找所有 name 以 'letter' 结尾的 input 元素
[selector1][selector2][selectorN] $("input[id][name$='man']") 找到所有含有 id 属性，并且它的 name 属性是以 man 结尾的
:first-child $("ul li:first-child") 在每个 ul 中查找第一个 li
:last-child $("ul li:last-child") 在每个 ul 中查找最后一个 li
:nth-child $("ul li:nth-child(2)") 在每个 ul 查找第 2 个li(:nth-child从1开始的，而:eq()是从0算起的！)
